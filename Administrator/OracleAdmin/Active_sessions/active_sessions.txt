select
		username,
		machine,
		count(*)
from
		gv$session
where
		type='USER'	
group by
		username,machine
order by 3 desc;

		inst_id,
		username,
		machine,

from
		gv$session
where
		type='USER'	
group by
		inst_id,username,machine
order by 4 desc;
